```csharp
 static void WriteArray(int[] Array, int index = 0)
 {
     Console.Write(Array[index] + "\t");
     index++;
     if (index >= Array.Length)
         return;
     WriteArray(Array, index);
 }

 static void Main(string[] args)
 {
     Console.WriteLine("введите длинну массива \n");
     int ammount = int.Parse(Console.ReadLine());

     Console.WriteLine("введите максимальное число в массиве \n");
     int maxNumber = int.Parse(Console.ReadLine());  

     Console.Clear();

     int[] myArray = new int[ammount];

     Random random = new Random();

     for (int i = 0; i < myArray.Length; i++)
     {
         myArray[i] = random.Next(maxNumber);
     }

     WriteArray(myArray);
```
