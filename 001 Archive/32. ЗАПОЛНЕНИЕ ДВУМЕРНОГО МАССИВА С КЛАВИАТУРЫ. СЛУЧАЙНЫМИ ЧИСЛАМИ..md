            Console.WriteLine("ВВедите высоту...");
            int height = int.Parse(Console.ReadLine());

            Console.WriteLine("\nВведите ширину...");

            int width = int.Parse(Console.ReadLine());


            int[,] myArray = new int[height, width];

            Random random = new Random();   

            for (int y = 0; y < height; y++)// Cначала заполняем, потом                 показываем
            {
                for (int x = 0; x < width; x++)
                {
                    myArray[y, x] = random.Next(100);
                }
                Console.WriteLine();
            }
            for (int y = 0; y < height; y++)//двигается по y
            {
                for (int x = 0; x < width; x++)//двигается по x
                {
                    Console.Write(myArray[y, x] + "\t");
                }
                Console.WriteLine();
            }
